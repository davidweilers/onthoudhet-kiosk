use iced::{Task, window};
use iced::widget::{Column, Text};
use std::time::Duration;

#[derive(Debug, Clone)]
enum Message {
    WindowOpened(window::Id),
    Screenshot,
    ScreenshotReady(window::Screenshot),
}

use iced::border;
use iced::keyboard;
use iced::mouse;
use iced::widget::{
    button, canvas, center, center_y, checkbox, column, container, pick_list, pin, responsive, row,
    rule, scrollable, space, stack, text,
};
use iced::{
    Center, Element, Fill, Font, Length, Point, Rectangle, Renderer, Shrink, Subscription, Theme,
    color,
};

pub fn main() -> iced::Result {
    iced::application(Layout::default, Layout::update, Layout::view)
    .run()
}

#[derive(Debug, Default)]
struct Layout {
    window_id: Option<window::Id>,
}

impl Layout {
    fn view(&self) -> iced::Element<Message> {
        column![
            text("Taking a screenshot...").size(30),
        ].into()
    }

    fn update(app: &mut Layout, message: Message) -> Task<Message> {
        match message {
            Message::WindowOpened(id) => {
                app.window_id = Some(id);
                Task::none()
            }

            Message::Screenshot => {
                if let Some(id) = app.window_id {
                    window::screenshot(id).map(Message::ScreenshotReady)
                } else {
                    Task::none()
                }
            }

            Message::ScreenshotReady(screenshot) => {
                let img = image::RgbaImage::from_raw(
                    screenshot.size.width,
                    screenshot.size.height,
                    screenshot.rgba.to_vec(),
                )
                .unwrap();

                img.save("screenshot.png").unwrap();

                std::process::exit(0);
            }
        }
    }
}


use iced::keyboard;
use iced::time::{self, Duration, Instant, milliseconds};
use iced::widget::{button, center, column, row, text};
use iced::window::Action::Screenshot;
use iced::{Center, Element, Subscription};
use iced::{Task, window};

pub fn main() -> iced::Result {
    let settings = window::Settings {
        ..Default::default()
    };

    iced::application(Stopwatch::default, Stopwatch::update, Stopwatch::view)
        .subscription(Stopwatch::subscription)
        // .window(settings)
        .run()
}

#[derive(Default, Debug)]
struct Stopwatch {
    duration: Duration,
    state: State,
    screenshot: Option<(Screenshot, image::Handle)>,
}

#[derive(Default, Debug)]
enum State {
    #[default]
    Idle,
    Ticking {
        last_tick: Instant,
    },
}

#[derive(Debug, Clone)]
enum Message {
    Init,
    Toggle,
    Reset,
    Tick(Instant),
    Screenshot,
    Screenshotted(Screenshot),
    ScreenshotReady(window::Screenshot),
}

impl Stopwatch {
    fn default() -> Self {
        
        Self {
            duration: Duration::default(),
            state: State::Idle,
            window_id: Some(window::Id::unique()),
        }
    }
    fn update(&mut self, message: Message) -> Task<Message> {
        match message {
            Message::Init => {
                println!("Initializing...{}", self.window_id.is_none());
                // if let self.window_id = Some(window::Id()) {
                //     println!("Window ID: {:?}", self.window_id);
                // } else {
                //     println!("Failed to get window ID");
                // }
                self.state = State::Ticking { last_tick: Instant::now() };
                Task::none()
            }
            Message::Toggle => match self.state {
                State::Idle => {
                    self.state = State::Ticking {
                        last_tick: Instant::now(),
                    };
                    Task::none()
                }
                State::Ticking { .. } => {
                    self.state = State::Idle;
                    Task::none()
                }
            },
            Message::Tick(now) => {
                if let State::Ticking { last_tick } = &mut self.state {
                    self.duration += now - *last_tick;
                    *last_tick = now;

                    println!("Tick: {:?}", self.duration);
                    if self.duration.as_secs() >= 2 {
                        println!("Taking screenshot...");
                         self.state = State::Idle;
                        println!("{:?}", self.window_id);
                        if let Some(id) = self.window_id {
                            return window::screenshot(id).map(Message::ScreenshotReady);
                        }
                    }
                }
                Task::none()
            }
            Message::Reset => {
                self.duration = Duration::default();
                Task::none()
            }
            Message::Screenshot => {
                return window::latest()
                    .and_then(window::screenshot)
                    .map(Message::Screenshotted);
            }
            Message::Png => {
                if let Some((screenshot, _handle)) = &self.screenshot {
                    self.png_saving = true;

                    return Task::perform(save_to_png(screenshot.clone()), Message::PngSaved);
                }
            }
            Message::PngSaved(res) => {
                self.png_saving = false;
                self.saved_png_path = Some(res);
            }
            Message::Screenshotted(screenshot) => {
                self.screenshot = Some((
                    screenshot.clone(),
                    image::Handle::from_rgba(
                        screenshot.size.width,
                        screenshot.size.height,
                        screenshot.rgba,
                    ),
                ));
            }
            Message::ScreenshotReady(screenshot) => {
                let img = image::RgbaImage::from_raw(
                    screenshot.size.width,
                    screenshot.size.height,
                    screenshot.rgba.to_vec(),
                )
                .unwrap();

                img.save("screenshot.png").unwrap();

                std::process::exit(0);
            }
        }
    }

    fn subscription(&self) -> Subscription<Message> {
        let tick = match self.state {
            State::Idle => Subscription::none(),
            State::Ticking { .. } => time::every(milliseconds(100)).map(Message::Tick),
        };

        fn handle_hotkey(event: keyboard::Event) -> Option<Message> {
            use keyboard::key;

            let keyboard::Event::KeyPressed { modified_key, .. } = event else {
                return None;
            };

            match modified_key.as_ref() {
                keyboard::Key::Named(key::Named::Space) => Some(Message::Toggle),
                keyboard::Key::Character("r") => Some(Message::Reset),
                _ => None,
            }
        }

        Subscription::batch(vec![tick, keyboard::listen().filter_map(handle_hotkey)])
    }

    fn view(&self) -> Element<'_, Message> {
        const MINUTE: u64 = 60;
        const HOUR: u64 = 60 * MINUTE;

        let seconds = self.duration.as_secs();

        let duration = text!(
            "{:0>2}:{:0>2}:{:0>2}.{:0>2}",
            seconds / HOUR,
            (seconds % HOUR) / MINUTE,
            seconds % MINUTE,
            self.duration.subsec_millis() / 10,
        )
        .size(40);

        let button = |label| button(text(label).align_x(Center)).padding(10).width(80);

        let toggle_button = {
            let label = match self.state {
                State::Idle => "Start",
                State::Ticking { .. } => "Stop",
            };

            button(label).on_press(Message::Toggle)
        };

        let reset_button = button("Reset")
            .style(button::danger)
            .on_press(Message::Reset);

        let controls = row![toggle_button, reset_button].spacing(20);

        let content = column![duration, controls].align_x(Center).spacing(20);

        center(content).into()
    }
}
